public class EjerciciosRecursivos {

    // EJERCICIO 1: Sumar d√≠gitos recursivamente
    public static int sumarDigitos(int n) {
        // Caso base: si n es menor a 10, ya no se puede dividir m√°s
        if (n < 10) {
            return n;
        }
        // Caso recursivo: se suma el √∫ltimo d√≠gito (n % 10) + suma de los d√≠gitos restantes
        return (n % 10) + sumarDigitos(n / 10);
    }

    // EJERCICIO 2.1: Verificar si una cadena es un pal√≠ndromo usando recursividad
    public static boolean esPalindromo(String s) {
        // Si la cadena tiene 0 o 1 caracteres, es un pal√≠ndromo
        if (s.length() <= 1) {
            return true;
        }
        // Comparar primer y √∫ltimo car√°cter
        if (s.charAt(0) != s.charAt(s.length() - 1)) {
            return false;
        }
        // Llamada recursiva eliminando el primer y √∫ltimo car√°cter
        return esPalindromo(s.substring(1, s.length() - 1));
    }

    // EJERCICIO 2.2: Calcular potencia de forma recursiva
    public static int potencia(int base, int exponente) {
        // Caso base: cualquier n√∫mero elevado a 0 es 1
        if (exponente == 0) {
            return 1;
        }
        // Caso recursivo: base * potencia(base, exponente - 1)
        return base * potencia(base, exponente - 1);
    }

    // EJERCICIO 2.3: Invertir una cadena recursivamente
    public static String revertir(String s) {
        // Caso base: cadena vac√≠a o de un solo car√°cter
        if (s.length() <= 1) {
            return s;
        }
        // Tomar el √∫ltimo car√°cter + revertir el resto de la cadena
        return s.charAt(s.length() - 1) + revertir(s.substring(0, s.length() - 1));
    }

    // M√©todo para calcular la distancia total de una ruta
    public static int calcularDistancia(int[][] distancias, int[] ruta) {
        int distanciaTotal = 0;
        for (int i = 0; i < ruta.length - 1; i++) {
            distanciaTotal += distancias[ruta[i]][ruta[i + 1]];
        }
        // Agregar la distancia de regreso a la ciudad de origen

        if (l == r) {
            // Calcular la distancia de la ruta actual
            int distanciaActual = calcularDistancia(distancias, ruta);

public class BusquedaBinaria {

    // M√©todo para realizar la b√∫squeda binaria
    public static int busquedaBinaria(int[] arr, int valorBuscado) {
        int inicio = 0;
        int fin = arr.length - 1;

        while (inicio <= fin) {
            int medio = inicio + (fin - inicio) / 2;  // Encontrar el √≠ndice medio

            // Verificar si el valor buscado est√° en el medio
            if (arr[medio] == valorBuscado) {
                return medio;  // Elemento encontrado
            }

            // Si el valor buscado es mayor, ignorar la mitad izquierda
            if (arr[medio] < valorBuscado) {
                inicio = medio + 1;
            } else {
                // Si el valor buscado es menor, ignorar la mitad derecha
                fin = medio - 1;
            }
        }

        // El valor no est√° presente en el arreglo
        return -1;
    }

    public static void main(String[] args) {
        int[] arreglo = {2, 3, 4, 10, 40};  // El arreglo debe estar ordenado
        int valorBuscado = 10;

        // Llamar al m√©todo de b√∫squeda binaria
